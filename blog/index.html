      # 安装和搭建

      # 全局安装 VuePress

      yarn global add vuepress 或者：npm install -g vuepress

      # 创建项目文件夹

      mkdir myBlog

      # 进入myBlog 文件夹， 初始化项目

      cd myBlog
npm init

      # 创建文件夹和文件

      在 myBlog 文件夹中创建 docs 文件夹， 在 docs 中创建 .vuepress 文件夹， 在 .vuepress 中创建 public 文件夹和 config.js 文件， 最终的项目结构如下：

      vuepressBlogDemo
    ├─── docs
    │   ├── .vuepress
    │   │   ├── public
    │   │   └── config.js
    │   ├── blog
    │   │   └── README.md
    │   ├── notes
    │   │   └── README.md
    │   └── README.md    
    └── package.json

      # 在 config.js 文件中配置网站的标题、描述、主题等信息

      module.exports = {
    permalink: '/:year/:month/:day:/slug',
    title: 'zcy\'s blog', //标题
    description: '我的个人网站', //副标题
    head: [ // 注入到当前页面的 HTML <head> 中的标签
      [ 'link', { rel: 'icon', href: '/logo.ico' }], // 增加一个自定义的 favicon
      [ 'link', { rel: 'manifest', href: '/logo.ico' }],
      [ 'link', { rel: 'apple-touch-icon', href: '/logo.ico' }]
    ],
    evergreen: true, // 浏览器兼容性 
    base: '/', // 这是部署到github相关的配置，部署到github仓库的分支选择
    markdown: {
      lineNumbers: true, // 代码块显示行号
      externalLinks: { target: '_black', rel: 'nooper noreferrer'}
    },
    themeConfig: {
      nav:[ // 导航栏配置
        { text: '搭建博客', link: '/blog/' },
        { text: '学习笔记',link: '/notes/' },
        { text: 'Github', link: 'https://github.com/younggga/blog'}      
      ],
      lastUpdated: 'Last Updated',
      sidebar: 'auto', // 侧边栏配置
      sidebarDepth: 2, // 侧边栏显示2级
    }
};

      # 部署到服务器

      # 登陆 Github

      打开 github 网站，登陆自己的 github 账号

      接着我们新建仓库： 新建仓库一： USERNAME.github.io

      ！！！注意：USERNAME 必须是 Github 的账号名称，要保证和Github账号名一模一样！

      # 在项目中，创建一个如下的 deploy.sh 文件

          #!/usr/bin/env sh

    # 确保脚本抛出遇到的错误
    set -e

    # 推送源码到git仓库(仓库二)
    # git init
    # git add -A
    # git commit -m 'deploy'
    # git push -f <仓库二地址>

    # 生成静态文件
    npm run build

    # 进入生成的文件夹
    cd docs/.vuepress/dist

    # 如果是发布到自定义域名
    # echo 'www.example.com' > CNAME

    git init
    git add -A
    git commit -m 'deploy'

    # 如果你想要部署到 https://USERNAME.github.io
    # git push -f git@github.com:USERNAME/USERNAME.github.io.git master

    # 如果发布到 https://USERNAME.github.io/<REPO>  REPO=github上的项目
    # git push -f git@github.com:USERNAME/<REPO>.git master:gh-pages

    cd -
